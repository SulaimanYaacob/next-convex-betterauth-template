      },
    });

    import { cronJobs } from "convex/server";
    import { internal } from "./_generated/api";
    import { internalMutation } from "./_generated/server";

    // PRES-03: every 1 minute, mark presence rows with lastSeen > 5 min ago as "offline".
    // Handles tab crashes (no client to send heartbeat). Per D-12.
    export const markStalePresence = internalMutation({
      args: {},
      handler: async (ctx) => {
        const STALE_MS = 5 * 60 * 1000;
        const cutoff = Date.now() - STALE_MS;
        // Use .take(100) per Convex guidelines — never .collect()
        const rows = await ctx.db.query("presence").take(100);
        for (const row of rows) {
          if (row.lastSeen < cutoff && row.status !== "offline") {
            await ctx.db.patch(row._id, { status: "offline" });
          }
        }
      },
    });

    const crons = cronJobs();
    crons.interval(
      "mark-stale-presence",
      { minutes: 1 },
      internal.crons.markStalePresence,
      {},
    );
    export default crons;

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser client → Convex queries (`getBalance`, `getOnlinePlayers`) | Client invokes public queries; identity must be derived server-side, not from arguments |
| Convex scheduler → internal mutation (`markStalePresence`) | Internal-only invocation surface; not callable from public API |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | S (Spoofing) | `coinLedger.getBalance` | mitigate | Server derives identity via `betterAuthComponent.getAuthUser(ctx)`; query takes no `userId` argument; impossible to read another user's balance |
| T-02-02 | T (Tampering) | `coinLedger.getBalance` | mitigate | Read-only query; cannot mutate ledger. All ledger writes are out-of-scope for Phase 2 (Phase 3+) |
| T-02-03 | I (Information Disclosure) | `presence.getOnlinePlayers` | accept | Returns first-2-letter initials + email-derived name (low PII). Online status is the inherent purpose of a public presence panel |
| T-02-04 | I (Information Disclosure) | `presence.getOnlinePlayers` | mitigate | Output strips raw email and userId-internal fields; only `userId` (Convex Id), `name`, `initials`, `status` exposed. No password / auth tokens leaked |
| T-02-05 | E (Elevation of Privilege) | `markStalePresence` | mitigate | Registered as `internalMutation` — not in `api.*`, only callable by Convex scheduler / other internal functions; no public surface |
| T-02-06 | D (Denial of Service) | `getOnlinePlayers`, `markStalePresence`, `getBalance` | mitigate | All queries bounded by `.take(50)` / `.take(100)` / `.take(1000)`; no `.collect()`. Cron runs every 1 minute (bounded fan-out) |
| T-02-07 | T (Tampering) | `presence` table status field | mitigate | Schema validator union limits status to 4 literals; cron's write of `"offline"` is now schema-valid (Task 1); no arbitrary status strings accepted |
| T-02-08 | R (Repudiation) | All queries | accept | Read-only; no write actions to repudiate. Cron writes are logged in Convex function logs |
</threat_model>
